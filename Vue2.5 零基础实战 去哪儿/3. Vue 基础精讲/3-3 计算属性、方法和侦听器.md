## 计算属性

> 它是内置缓存的，所依赖变量不变时，属性也不会改变。

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <title>Vue 实例生命周期函数</title>
    <link rel="stylesheet" href="">
    <script src="https://cdn.bootcss.com/vue/2.5.17/vue.js"></script>
</head>
<body>
    <div id="app">
        {{ fullName }}
        {{ age }}
    </div>


    <script>
        var vm = new Vue({
            el: '#app',
            data: {
                firstName: 'Dell',
                lastName: 'Lee',
                age: 28
            },
            computed: {
                fullName: function() {
                    console.log('计算了一次');
                    return this.firstName + " " + this.lastName
                }
            }
        })
    </script>
</body>
</html>  
```



## 方法

> 方法是没有内置缓存的

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <title>Vue 实例生命周期函数</title>
    <link rel="stylesheet" href="">
    <script src="https://cdn.bootcss.com/vue/2.5.17/vue.js"></script>
</head>
<body>
    <div id="app">
        {{ fullName() }}
        {{ age }}
    </div>


    <script>
        var vm = new Vue({
            el: '#app',
            data: {
                firstName: 'Dell',
                lastName: 'Lee',
                age: 28
            },
            methods: {
                fullName: function() {
                    return this.firstName + " " + this.lastName
                }
            }
        })
    </script>
</body>
</html>  
```



## 侦听

> 带有缓存机制

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <title>Vue 实例生命周期函数</title>
    <link rel="stylesheet" href="">
    <script src="https://cdn.bootcss.com/vue/2.5.17/vue.js"></script>
</head>
<body>
    <div id="app">
        {{ fullName }}
        {{ age }}
    </div>


    <script>
        var vm = new Vue({
            el: '#app',
            data: {
                firstName: 'Dell',
                lastName: 'Lee',
                fullName: 'Dell Lee',
                age: 28
            },
            watch: {
                firstName: function() {
                    this.fullName = this.firstName + " " + this.lastName;
                },
                lastName: function() {
                    this.fullName = this.firstName + " " + this.lastName;
                }
            }
        })
    </script>
</body>
</html>  
```
